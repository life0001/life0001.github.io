示例1：单日历

var calendar = new Calendar({
    id: ['calendar'],
    outputId: ['a']
});
calendar.init();

示例2：单日历，指定年月日的日历

var calendar1 = new Calendar({
            id: ['calendar2'],
            outputId: ['b'],
            targetDate: 20290305
        });
calendar1.init();

示例3：耦合双日历

var calendar2 = new Calendar({
            id: ['calendar3', 'calendar4'],
            outputId: ['c', 'd']
        });
calendar2.init();
